///general operation//
- show plane status (p3/p6 current position&velocity)
command: show plane_id status
	argurment: plane_id
	output: plane_id, x,y,z, speed_x,speed_y,speed_z


- add an aircraft object (p6)
command: add aircraft plane_id, speed_x,speed_y,speed_z,x,y,z,entrytime
  arguement: plane_id, speed_x, speed_y, speed_z,x,y,z,entrytime
  output: "plane_id added"

- remove an aircraft object (p6)
command: delete aircraft plane_id
  arguement: plane_id
  output: "plane_id deleted"

- display data record for a given aircraft (p6)
command: display plane_id data record
  arguement: plane_id
  output: data record for plane_id from entry time to current time

- project plane position to current_time + n (p6)
command: show plane_id position AddTime
   arguement: plane_id, AddTime
   output: plane_id, x,y,z, "at time current_time+AddTime"//x,y,z at current_time+AddTime
///end general operation//
